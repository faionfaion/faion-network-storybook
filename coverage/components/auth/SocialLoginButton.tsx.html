const ProviderNames: Record<SocialProvider, string> = {
  google: 'Google',
  github: 'GitHub',
  linkedin: 'LinkedIn',
  twitter: 'X',
  apple: 'Apple',
  microsoft: 'Microsoft',
};
 
/**
 * SocialLoginButton - Social authentication button
 *
 * Buttons for Google, GitHub, LinkedIn, etc. sign-in.
 * Follows the "Digital Journal" theme.
 */
export default function SocialLoginButton({
  provider,
  label,
  iconOnly = false,
  isLoading = false,
  fullWidth = false,
  disabled,
  className = '',
  ...props
}: SocialLoginButtonProps) {
  const buttonLabel = label || `Continue with ${ProviderNames[provider]}`;
 
  return (
    <button
      className={`fn-social-btn fn-social-btn--${provider} ${iconOnly ? 'fn-social-btn--icon-only' : ''} ${fullWidth ? 'fn-social-btn--full-width' : ''} ${className}`}
      disabled={disabled || isLoading}
      aria-label={buttonLabel}
      {...props}
    >
      {isLoading ? (
        <span className="fn-social-btn__spinner" />
      ) : (
        <>
          <span className="fn-social-btn__icon">{ProviderIcons[provider]}</span>
          {!iconOnly && <span className="fn-social-btn__text">{buttonLabel}</span>}
        </>
      )}
    </button>
  );
}
 
// Named export
export { SocialLoginButton };
